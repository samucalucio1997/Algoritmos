#include <stdio.h>
void primo(int n){
    if(n==1){
        printf("Não");
    }
    if(n==2){
        printf("Sim");
    }
    for(int i=2;i<n;i++){
        if(n%i==0 && n>2){
            printf("Não");
            break;
        }
        if(i==n-1){
         printf("Sim");
        }    
    }
}
int main(){
    int entrada;
    scanf("%d",&entrada);
    primo(entrada);
    return 0;
}
